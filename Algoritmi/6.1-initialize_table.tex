\begin{algorithm}[H] \floatname{algorithm}{Algoritmo}
    \caption{INITIALIZE-TABLE($V$)}\label{alg:init-table}
    \begin{algorithmic}[1]
        \State Sia $T$ una nuova tabella
        \For{$v \in V$}
            \State $T[v] \coloneqq \emptyset$
        \EndFor
        \State \Return $T$
    \end{algorithmic}
\end{algorithm}
