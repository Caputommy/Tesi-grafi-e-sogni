\begin{algorithm}[H] \floatname{algorithm}{Algoritmo}
    \caption{ADD-SINGLETONS($T$)}\label{alg:add-singletons}
    \begin{algorithmic}[1]
        \For{$v \in T.keys$}
            \If{$T[v] == \emptyset$}
                \State $T[v] := \{v\}$
            \EndIf
        \EndFor
    \end{algorithmic}
\end{algorithm}
